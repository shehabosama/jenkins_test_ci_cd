pipeline {
    agent any

    stages {
        stage('Checkout') {
            steps {
                // Checkout the code from the Git repository
                git branch: 'master', url: 'https://github.com/your_username/your_android_project.git'
            }
        }

        stage('Build') {
            steps {
                // Build the Android project
                sh './gradlew assembleDebug'
            }
        }

        stage('Run Tests') {
            steps {
                // Run unit tests for the Android project
                sh './gradlew test'
            }
        }

        stage('Archive APK') {
            steps {
                // Archive the generated APK file
                archiveArtifacts artifacts: 'app/build/outputs/apk/debug/*.apk'
            }
        }
    }

    post {
        always {
            // Clean up workspace after the build
            cleanWs()
        }
    }
}
